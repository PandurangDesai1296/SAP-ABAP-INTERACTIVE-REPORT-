        DOB     TYPE ZSTUDENT_DATA-DOB,
        GENDER  TYPE ZSTUDENT_DATA-GENDER,

        PHONE   TYPE ZSTUDENT_DATA-PHONE,

        FEES        TYPE ZCOURSE_DATA-FEES,

        FAC_ID        TYPE ZFACULTY_DATA-FAC_ID,
        FA_COURSE_ID  TYPE ZFACULTY_DATA-COURSE_ID,
        SUBJECT       TYPE ZFACULTY_DATA-SUBJECT,
        FA_EMAIL      TYPE ZFACULTY_DATA-EMAIL,

        ENROLL_ID     TYPE ZENROLLMENT_DATA-ENROLL_ID,
        EN_STUD_ID    TYPE ZENROLLMENT_DATA-STUD_ID,
        EN_COURSE_ID  TYPE ZENROLLMENT_DATA-COURSE_ID,
        ENROLL_DATE   TYPE ZENROLLMENT_DATA-ENROLL_DATE,

  END OF ST_COLLEGE.

DATA : IT_COLLEGE TYPE TABLE OF ST_COLLEGE.
DATA : WA_COLLEGE TYPE ST_COLLEGE.

DATA : IT_FCAT TYPE LVC_T_FCAT.
DATA : WA_FCAT TYPE LVC_S_FCAT.

SELECT-OPTIONS : S_ID FOR ZSTUDENT_DATA-STU_ID.

START-OF-SELECTION.

SELECT    A~STU_ID
          A~NAME
          A~DOB
          A~GENDER
          A~EMAIL
          A~PHONE
          A~ADDRESS

          B~COURSE_ID
          B~STU_ID  AS CU_STU_ID
          B~COURSE_NAME
          B~DURATION
          B~FEES

          D~FAC_ID
          D~COURSE_ID AS FA_COURSE_ID
          D~SUBJECT
          D~EMAIL AS FA_EMAIL

          F~ENROLL_ID
          F~STUD_ID   AS EN_STUD_ID
          F~COURSE_ID AS EN_COURSE_ID
          F~ENROLL_DATE

INTO TABLE IT_COLLEGE
FROM ZSTUDENT_DATA AS A
  INNER JOIN ZCOURSE_DATA AS B
  ON A~STU_ID = B~STU_ID

  LEFT OUTER JOIN ZFACULTY_DATA AS D
  ON B~COURSE_ID = D~COURSE_ID

  INNER JOIN ZENROLLMENT_DATA AS F
  ON A~STU_ID = F~STUD_ID

WHERE A~STU_ID IN S_ID.
